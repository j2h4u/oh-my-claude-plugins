<objective>
Replace the per-call `_db()` pattern with a lazy singleton connection so that all cache reads/writes within a single statusline render reuse one SQLite connection, eliminating redundant PRAGMA and DDL execution on every call.

Purpose: Each render currently opens 8-12 separate SQLite connections (one per `cache_get`/`cache_put` call), each running PRAGMA journal_mode=WAL and CREATE TABLE IF NOT EXISTS. This is pure overhead -- the connection can be opened once and reused for the entire ~50ms render lifetime.

Output: Modified `omcc-statusline.py` with singleton connection pattern.
</objective>

```python
def _db() -> sqlite3.Connection:
    """Open cache DB (WAL mode, short busy timeout for readers)."""
    CACHE_DIR.mkdir(parents=True, exist_ok=True)
    con = sqlite3.connect(str(CACHE_DB), timeout=2)
    con.execute("PRAGMA journal_mode=WAL")
    con.execute(
        "CREATE TABLE IF NOT EXISTS cache "
        "(key TEXT PRIMARY KEY, data TEXT NOT NULL DEFAULT '{}',"
        " updated_at REAL NOT NULL DEFAULT 0,"
        " cooldown_until REAL NOT NULL DEFAULT 0)"
    )
    return con

def cache_get(key: str) -> tuple[str | None, float, float]:
    """Return (data, updated_at, cooldown_until) for a cache key."""
    try:
        con = _db()
        row = con.execute(
            "SELECT data, updated_at, cooldown_until FROM cache WHERE key = ?", (key,)
        ).fetchone()
        con.close()
        if row:
            return row[0], row[1], row[2]
    except sqlite3.Error:
        pass
    return None, 0.0, 0.0

def cache_put(key: str, data: str, cooldown_until: float = 0.0) -> None:
    """Write data to cache (from main process)."""
    try:
        con = _db()
        con.execute(
            "INSERT OR REPLACE INTO cache (key, data, updated_at, cooldown_until)"
            " VALUES (?, ?, ?, ?)",
            (key, data, time.time(), cooldown_until),
        )
        con.commit()
        con.close()
    except sqlite3.Error:
        pass
```
